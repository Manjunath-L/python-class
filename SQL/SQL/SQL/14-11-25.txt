  TO REPLACE SUBSTRING FROM NEW STRING IN ORIGINAL STRING.

SYNTAX:
       REPLACE('ORIGINAL_STR',SUB_STR,NEW_STR);
 
     EX: SELECT REPLACE('PENTAGON','PENT','HEX');//| HEXAGON
