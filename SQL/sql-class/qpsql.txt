1. Find employees whose salary is greater than the minimum salary in the company.
SELECT *
FROM emp
WHERE SAL > (SELECT MIN(SAL)
            FROM EMP);

2. List employees working in the department with the highest department NO.
SELECT *
FROM emp
WHERE DNO = (
    SELECT MAX(DNO)
    FROM EMP
);

3. Find employees whose salary is less than the average salary.
SELECT *
FROM emp
WHERE SAL < (
    SELECT AVG(SAL)
    FROM EMP
);
