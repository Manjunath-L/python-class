CID     INT UNSIGNED NN   PRIMARY KEY AUTO_ICREMENT
CNAME  VARCHAR(20) NN
PHONE  BIGINT UNSIGNED NN UNIQUE,CHECK(LENGTH(PHONE)=10)
GENDER ENUM('MALE','FEMALE') NN
BALANCE DECIMAL(5,2) UNSIGNED NN DEFAULT 100



CREATE TABLE CUSTOMER
(
  CID INT UNSIGNED NOT NULL PRIMARY KEY AUTO_INCREMENT,
 CNAME VARCHAR(20) NOT NULL,
 PHONE BIGINT UNSIGNED NOT NULL UNIQUE,CHECK(LENGTH(PHONE)=10),
 GENDER ENUM('MALE','FEMALE'),
 BALANCE DECIMAL(5,2) UNSIGNED NOT NULL DEFAULT'100',
 SID INT UNSIGNED NOT NULL,
 CONSTRAINT FOREIGN KEY(SID)REFERENCES STUDENT(SID)
);
